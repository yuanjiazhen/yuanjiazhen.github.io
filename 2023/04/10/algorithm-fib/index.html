经典问题-菲波那契数列

Jia

2023-04-10 21:02:15

定义：菲波那切数列：F (0)=0， F (1)=1, F (n)= F (n - 1)+ F (n - 2)（ n ≥ 2， n ∈ N*）
举例：0、1、1、2、3、5、8、13、21…(n-2)+(n-1)
解决方案：递归、递归+记忆化、动态规划、动态规划+滚动数组

递归

// 自顶向下，层层递归，O(2^n)
function fib(n) {
  if (n <= 1) return n;
  return fib(n - 2) + fib(n - 1);
}
console.log("递归：", fib(8)); // 21

递归+记忆化

// 递归+记忆化，空间换时间，减少重复计算
function fib1(n) {
  const memo = {}; // 对已计算出的结果存储起来

  const helper = (n) => {
    if (memo[n]) return memo[n]; // 已计算过直接返回，不重复计算
    if (n <= 1) return n;
    const result = helper(n - 2) + helper(n - 1);
    memo[n] = result;
    return result;
  };
  return helper(n);
}
console.log("递归+记忆化：", fib1(8)); // 21

动态规划

// 动态规划：下一个问题的解依赖于上一个子问题的解
function fib2(n) {
  if (n <= 1) return n;

  const result = [0, 1]; // 直接存 n <=1 的结果
  for (let i = 2; i <= n; i++) {
    result[i] = result[i - 2] + result[i - 1];
  }
  return result[n];
}
console.log("动态规划：", fib2(8)); // 21

动态规划+滚动数组

// 动态规划+滚动数组：节省存储空间
function fib3(n) {
  if (n <= 1) return n;

  let result = [0, 1]; // 直接存 n <=1 的结果
  for (let i = 2; i <= n; i++) {
    const sum = result[0] + result[1];
    result = [result[1], sum];
  }
  return result[1];
}
console.log("动态规划+滚动数组：", fib3(8)); // 21
