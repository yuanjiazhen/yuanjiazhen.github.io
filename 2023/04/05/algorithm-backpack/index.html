经典问题-背包问题

Jia

2023-04-05 20:30:01

问题描述：给定一个固定大小、能够携带重量 W 的背包，以及一组有价值和重量的物品，找出一个最佳解决方案，使得装入背包的物品总重量不超过 W 且总价值最大；只能往背包中装完整的物品

解决方案：采用动态规划，将一个大问题拆成一个个小问题，通过解决小问题得到大问题的答案

/*
 * 背包问题：只能往背包装完整物品
 * 动态规划
 * weights:  物品重量；values：物品价值；w 背包能装最大重量
 */
function knapsack(weights, values, maxWeight) {
  let result = []
  for (let w = 0; w < weights.length; w++) {
    for (let i = 1; i <= maxWeight; i++) {
      const index = i - 1
      if (!result[w]) {
        // new line
        result[w] = []
      }

      if (weights[w] <= i) {
        // 背包能装下
        if (!w) {
          // 第一行，特殊处理，因为有也多余容量也只能取本身
          result[w][index] = values[w]
          continue
        }

        const overWeight = i - weights[w]
        const value = overWeight > 0 ? (values[w] + result[w - 1][overWeight - 1]) : values[w]
        result[w][index] = Math.max(result[w - 1][index], value)
      } else {
        // 单个重量超重了，装不下
        result[w][index] = w ? result[w - 1][index] : 0
      }
    }
  }

  console.log(result, 'resultresult');
  return result[weights.length - 1][maxWeight - 1];
}

// console.log('result:', knapsack([1, 4, 3], [1500, 3000, 2000], 4));
console.log('result:', knapsack([2, 2, 6, 5, 4], [6, 3, 5, 4, 6], 10));
