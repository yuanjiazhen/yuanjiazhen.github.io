Vue 响应式系统

Jia

2022-11-10 21:35:11

响应式数据

响应式数据就是当数据改变后，可以通知到与这数据有关联的副作用函数执行，如下：

const obj = { text: 'hello world' }
function effct() {
  document.body.innerHTML = obj.text
}
// 响应式数据 obj：期望修改 obj.text 副作用函数 effct()可重新执行，更新视图 UI 为 'hello vue'
obj.text = 'hello vue'

手写响应式-基础版

利用 Proxy 可以创建对象的代理，拦截对象的 get、set操作

  • get：读操作时，存储副作用函数
  • set：取操作时，执行副作用函数
const bucket = new Set() // 存储副作用函数
// reactive 函数作用：将 data 包装成响应式数据
const reactive = (data) => {
  const reactiveData = new Proxy(data, {
    get(target, key) {
      // 读取 data 时，存储与 data 关联的副作用函数 effct
      // 基础版 effct 采用硬编码方式，写死名称
      bucket.add(effct)
      return target[key]
    },
    set(target, key, newVal) {
      target[key] = newVal
      // 修改 data 值时，从 bucket 取出副作用函数依次执行
      bucket.forEach(fn => fn())
    },
  })
  return reactiveData
}

// 测试
const obj = reactive({ text: 'hello world' })
function effct() {
  document.body.innerHTML = obj.text
}
effct()
// 1s 后修改 obj.text 值，期望视图由 'hello world' 变成 'hello vue'
setTimeout(() => {
  obj.text = 'hello vue'
}, 1000)

手写响应式-注册副作用函数

“基础版”的响应式的副作用函数effct是硬编码的，如果修改effct的名字，代码就如法正常运行，那么我们就需要一个注册副作用函数的机制，支持任何命名的副作用函数：

  • 定义activeEffect存储任何命名的副作用函数
  • 执行fn触发数据读操作，将fn存储在bucket中
  • 数据写操作时，将bucket中副作用函数拿出来执行
const bucket = new Set()
let activeEffect; // 【新增】存储注册的副作用函数：解决基础版硬编码“effect”
// 【新增】用来注册副作用函数
function effect(fn) {
  activeEffect = fn
  fn()  // 执行 fn，触发 reactive get 操作，存储 fn
}

const reactive = (data) => {
  const reactiveData = new Proxy(data, {
    get(target, key) {
      // 读取 data 时，存储与 data 关联的副作用函数 activeEffect 
      activeEffect && bucket.add(activeEffect)  // 【新增】
      return target[key]
    },
   // ...省略
  })
  return reactiveData
}
effect(changeBodyText)  // 【新增】用 effect 注册副作用函数

手写响应式-目标属性与副作用函数关联

“注册副作用函数”的响应式其实建立的是对象和副作用函数之间的关联，并未具体到对象属性和副作用函数的关联，这样就会引发只要修改对象上任一属性，都会触发副作用函数执行的问题

const obj = reactive1({ text: 'hello world' })
effect(()=>{
  console.log('effect 执行了')  // 会打印两次
  document.body.innerHTML = obj.text
})
obj.text = 'hello vue'  // 会触发 effect 执行
obj.desc = 'something'  // desc 未和 effect 关联也会触发 effect 执行

要解决effect被触发两次问题，只要修改响应式和effect的关联就可以了

const bucket = new WeakMap()
// ...省略 effect 相关代码
const reactive = (data) => {
  const reactiveData = new Proxy(data, {
    get(target, key) {
      track(target, key)   // 【新增】存储副作用函数
      return target[key]
    },
    set(target, key, newVal) {
      target[key] = newVal
      trigger(target, key) // 【新增】执行副作用函数
    },
  })
  return reactiveData
}
// get 拦截时调用 track 追踪变化
function track(target, key) {
  if (!activeEffect) return // 没副作用函数时，直接返回
  let depsMap = bucket.get(target)
  if (!depsMap) {
    // 建立 target-->key 关联
    bucket.set(target, depsMap = new Map())
  }
  let deps = depsMap.get(key)
  if (!deps) {
    // 建立 key-->effect 关联
    depsMap.set(key, deps = new Set())
  }
  // 添加 effect
  deps.add(activeEffect)
}
// set 拦截时调用 trigger 触发变化
function trigger(target, key) {
  // 读取 target 中的 key
  const depsMap = bucket.get(target)
  if (!depsMap) return
  // 读取 key 中关联的 effect
  const effcts = depsMap.get(key)
  effcts && effcts.forEach(fn => fn())
}

// 测试
const obj = reactive({ text: 'hello world' })
effect(()=>{
  console.log('effect 执行了')  // 会打印一次
  document.body.innerHTML = obj.text
})
obj.text = 'hello vue'  // 会触发 effect 执行
obj.desc = 'something'  // 不会触发 effect 执行

手写响应式-避免死循环

“注册副作用函数”的响应式会带来一个无限递归循环，造成栈溢出问题

  • 执行obj.foo触发obj的get操作，将副作用收集到桶中
  • 执行obj.foo = obj.foo + 1触发obj的set操作，执行桶中副作用函数，执行时又会触发obj.foo get操作，造成循环性的触发get``set操作，不断的执行副作用函数
const obj = reactive({ foo: 1 })
effect(() => obj.foo = obj.foo + 1) // 无限循环触发读写操作，操作栈溢出

解决方案：get操作执行完副作用函数后，重置activeEffect，在set触发副作用函数执行时，判断是否和现在正在执行的副作用函数相同，相同则不执行

let activeEffect;
function effect(fn) {
  activeEffect = fn
  fn()
  activeEffect = null  // 【新增】注册完副作用函数后，重置 activeEffect，避免一直占用
}

function trigger(target, key) {
  // ...省略
  const effcts = depsMap.get(key)
  effcts && effcts.forEach(fn => {
    // 【新增】当在 set 中要触发副作用函数时，如与当前正在执行的副作用函数相同，则不继续执行
    if (fn === activeEffect) return
    fn()
  })
}

手写响应式-支持调度

可调度：当set触发副作用函数重新执行时，可以调度副作用函数的执行时机、次数以及方式等，以执行时机为例

// 在未调度时，下面代码打印顺序为：1-2-end
// 期望调度执行时机后，印顺序为：1-end-2，即 obj.foo++ 会调度执行
const obj = reactive4({ foo: 1 })
effect(() => { console.log('obj.foo==>', obj.foo) })
obj.foo++
console.log('end')

解决方案：为effect函数添加一个可选配置参数options，用来设置调度相关的逻辑

let activeEffect;
function effect(fn, options = {}) {
  const effectFn = () => {   // 【新增】修改副作用函数结构，使其可挂载 options 属性
    activeEffect = effectFn
    fn()
    activeEffect = null
  }
  effectFn.options = options  // 【新增】将 options 添加到副作用函数中
  effectFn()
}
function trigger(target, key) {
  // ...省略
  effcts && effcts.forEach(fn => {
    if (fn === activeEffect3) return
    if (fn.options.scheduler) { // 【新增】当存在 options 时，单独处理
      console.log('进入调度了')
      fn.options.scheduler(fn)  // 将 fn 的执行权交给 scheduler，让用户自己决定执行时机
    } else {
      fn()  // 按之前逻辑默认执行
    }
  })
}
// 测试：期望在不改变代码顺序时，打印顺序由：1-2-end 变成 1-end-2，即 obj.foo++ 调度执行
const obj = reactive4({ foo: 1 })
effect(
  () => { console.log('obj.foo==>', obj.foo) },
  { // scheduler 调度副作用函数执行时机，也可以增加其它配置进行相应的处理
    scheduler(fn) {
      setTimeout(fn)  // 将 fn 放到宏任务队列中执行
    }
  }
)
obj.foo++
console.log('end')

手写响应式-完善非原始值代理

Proxy 是对对象的代理，可以拦截对象的基本操作和自定义，也就是说，Proxy 是不能代理非对象的（这种情况后面讨论），代理对象是只能拦截/自定义基本操作，如读取、设置，对于非基础操作（复杂操作），如对象下的方法调用是无法继续拦截的，这就会暴露出下面的问题：

const data = {
  foo: 1,
  get bar() {  // 这个为 getter
    // this 指向 data 原始对象, this.foo 等价于 data.foo，是不会建立副作用函数和 data.foo 的联系的
    return this.foo
  }
}
const obj = reactive(data)
// 正常流程：obj.bar 时执行 getter 函数，getter 访问了 this.foo，读取了 foo 的值，所以副作用函数会和 foo 建立联系，当修改 foo 是会触发副作用函数执行
effect(() => { 
  console.log('obj.foo==>', obj.bar) 
})
// 【期望】修改 foo 会触发副作用函数执行
// 【实际】修改 foo 不会触发副作用函数执行
obj.foo++

上述问题就出现在get bar(){ return this.foo }中，bar 的this其实指向的是data而非响应式obj所以obj.foo++是不会触发副作用函数执行的

这里就可以通过 Reflect 来解决，Reflect.get()，Reflect.set()的第三个参数是可以为getter指定this

const reactive = (data) => {
  const reactiveData = new Proxy(data, {
    get(target, key, receiver) {
      track(target, key)
      // return target[key]
      return Reflect.get(target, key,receiver)    // 【新增】

    },
    set(target, key, newVal, receiver) {
      // target[key] = newVal
      const res = Reflect.set(target, key,newVal,receiver)    // 【新增】
      trigger(target, key)
      return res
    },
  })
  return reactiveData
}

// 测试
const data = {
  foo: 1,
  get bar() {
    // 此时 this 指向调用 getter 的 obj
    return this.foo
  }
}
const obj = reactive(data)
effect(() => { 
  console.log('obj.foo==>', obj.bar) 
})
// 修改 foo 会触发副作用函数执行
obj.foo++

手写响应式-支持深响应

上述实现的reactive都是浅响应的，如下：

const obj = reactive({
  foo: {
    bar: 1
  }
})
effect(() => {
  console.log('obj.foo.bar==>', obj.foo.bar)
})
obj.foo.bar++ // 【问题】修改 obj.foo.bar 是不会触发响应的

原因是Reflect.get拿到的{ bar: 1 }是一个普通对象，并不会建立响应连接，如果需要建立深度响应连接，那么需要修改Reflect.get

const reactive = (data) => {
  const reactiveData = new Proxy(data, {
    get(target, key, receiver) {
      track(target, key)
      const res = Reflect.get(target, key, receiver)
      if (typeof res === 'object' && res !== null) {    // 【新增】不断递归处理深层响应
        return reactive(res)
      }
      return res
    },
    // ...省略
  })
  return reactiveData
}

修改后的reactive就可以支持深响应了，obj.foo.bar++修改时会触发响应

手写响应式-支持原始值

上述reactive响应式方案是借助Proxy实现的，Proxy是代理对象的，无法代理原理值（Boolean、Number、String、undefined、null、Symbol、BigInt 等），因为原始值是按值传递的，并非按引用传递，无法在形参和实参之间建立联系。如果想将原始值变成响应式数据，就需对其做一层包裹

// Proxy 是无法代理非对象值的，所以以下代码会报错
let age = reactive(1)
effect(() => {
  console.log('res ==>', age)
})
age++

引出ref将原始值包装成对象，使其可用Proxy代理，从而支持响应式：

function ref(val){
  const wrapper = {
    value: val
  }
  // 在 wrapper 上定义一个不可枚举属性 __v_isRef: true，用来区分是 ref 数据还是普通响应式数据
  Object.defineProperty(wrapper, '__v_isRef',{
    value: true
  })
  return reactive(wrapper)
}

// 测试
let age = ref(1)
effect(() => {
  console.log('res ==>', age.value)
})
age.value++ // 会触发响应式
